            #include <stdio.h>
            #include <math.h>
            int main() {
            int x, y, r;
            printf("Enter x, y, and radius: ");
            scanf("%d %d %d", &x, &y, &r);

            int distSq = x*x + y*y;
            int rSq = r*r;
            if (distSq < rSq)
                printf("Inside Circle\n");
            else if (distSq == rSq)
                printf("On the Circle\n");
            else
                printf("Outside Circle\n");

            return 0;
            }

            #include <stdio.h>
            int main() {
            double a, b;
            char op;
            printf("Enter first number, operator, second number: ");
            scanf("%lf %c %lf", &a, &op, &b);

            if (op == '+')
                printf("Result = %.2lf\n", a + b);
            else if (op == '-')
                printf("Result = %.2lf\n", a - b);
            else if (op == '*')
                printf("Result = %.2lf\n", a * b);
            else if (op == '/')
                printf("Result = %.2lf\n", b != 0 ? a / b : 0);
            else
                printf("Invalid Operator\n");

            return 0;}
