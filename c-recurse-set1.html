Answer:

A function that calls itself directly or indirectly, to solve a problem by breaking it down into smaller subproblems of the same type, until a base case is reached.

Key parts of any recursive function:


            1.Base Case –
                The condition where the function stops calling itself.(Prevents infinite recursion)
            2.	Recursive Case –
                The part where the function calls itself with a smaller or simpler version of the original problem.

General form in C:


                return_type function_name(parameters) {
                if (base_condition) {
                    // Base case
                    return some_value;
                } else {
                    // Recursive case
                    return function_name(modified_parameters);
                }
            }

               #include <stdio.h>
               #include <string.h>
                // Function to find the longest palindromic subsequence length (Recursion)
                int longestPalindromic(char str[], int i, int j, int count) {
                    // Base condition: when start index crosses end index
                    if (i > j)
                        return count;
                    // Base condition: when only one char remains
                    if (i == j)
                        return count + 1;
                    // If corner characters match
                    if (str[i] == str[j]) {
                        count = longestPalindromic(str, i + 1, j - 1, count + 2);
                        int option1 = longestPalindromic(str, i + 1, j, 0);
                        int option2 = longestPalindromic(str, i, j - 1, 0);    // Return the maximum among all possibilities
                        return (count > option1 ? (count > option2 ? count : option2) 
                                                : (option1 > option2 ? option1 : option2));
                    }
                    // If they don't match, skip one character from either side
                    int skipLeft = longestPalindromic(str, i + 1, j, 0);
                    int skipRight = longestPalindromic(str, i, j - 1, 0);
                    return (skipLeft > skipRight ? skipLeft : skipRight);
                }
                int main() {
                    char str[] = "aaaabbaa";
                    int n = strlen(str);
                    // Function call
                    int result = longestPalindromic(str, 0, n - 1, 0);
                    printf("%d\n", result);
                    return 0;
                }

                #include <stdio.h>
                // Function to print the Fibonacci series in reverse order
                void fibo(int n, int a, int b) {
                    if (n > 0) {
                        // Recursive call
                        fibo(n - 1, b, a + b);// Print after recursion so it comes in reverse order
                        printf("%d ", a);
                    }}
                int main() {
                    int N = 10;
                    fibo(N, 0, 1);
                    return 0;
                }

            #include <stdio.h>
            // Function to print a row of stars
            void printn(int num) {
                if (num == 0)
                    return;
                printf("* ");
                printn(num - 1);
            }
            // Function to print the pattern
            void pattern(int n, int i) {
                if (n == 0)
                    return;
                printn(i);
                printf("\n");
                pattern(n - 1, i + 1);
            }
            int main() {
                int n = 5;
                pattern(n, 1);
                return 0;
            }

                 #include <stdio.h>
                // Recursive function to calculate nCr
                int nCr(int n, int r) {
                    // Base cases
                    if (r == 0 || r == n)
                        return 1;
                    // Recursive case
                    return nCr(n - 1, r - 1) + nCr(n - 1, r);
                }
                int main() {
                    int N, r;
                    // Example inputs
                    N = 5; 
                    r = 2;
                    printf("%dC%d = %d\n", N, r, nCr(N, r));
                    N = 3; 
                    r = 1;
                    printf("%dC%d = %d\n", N, r, nCr(N, r));
                return 0;
