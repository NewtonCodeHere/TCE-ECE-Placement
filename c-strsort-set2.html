                #include <stdio.h>
                #include <string.h>
                int main() {
                    int n, i, j;
                    char str[20][100], temp[100];  // max 20 strings, each up to 100 chars
                    printf("Enter number of strings: ");
                    scanf("%d", &n);
                    printf("Enter %d strings:\n", n);
                    for(i = 0; i < n; i++) {
                        scanf("%s", str[i]);
                    }
                    // Bubble sort by string length
                    for(i = 0; i < n-1; i++) {
                        for(j = i+1; j < n; j++) {
                            if(strlen(str[i]) > strlen(str[j])) {
                                strcpy(temp, str[i]);
                                strcpy(str[i], str[j]);
                                strcpy(str[j], temp);
                            }
                        }
                    }
                    printf("\nStrings in order of length:\n");
                    for(i = 0; i < n; i++) {
                        printf("%s\n", str[i]);
                    }
                    return 0;}
