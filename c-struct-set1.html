            #include <stdio.h>
            struct Date { int d, m, y; };
            int compare(struct Date a, struct Date b) {
                if(a.y != b.y) return a.y - b.y;
                if(a.m != b.m) return a.m - b.m;
                return a.d - b.d;
            }
            int main() {
                struct Date d1 = {12, 8, 2025}, d2 = {11, 8, 2025};
                if(compare(d1, d2) > 0) 
                    printf("d1 is later\n");
                else if(compare(d1, d2) < 0) 
                    printf("d2 is later\n");
                else 
                    printf("Dates are equal\n");
                return 0;
            }
