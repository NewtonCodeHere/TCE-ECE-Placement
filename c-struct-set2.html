            #include <stdio.h>
            #include <math.h>

            struct Point3D {
                float x, y, z;
            };

            float distance(struct Point3D p1, struct Point3D p2) {
                return sqrt(pow(p2.x - p1.x, 2) +
                pow(p2.y - p1.y, 2) +
                pow(p2.z - p1.z, 2));
            }

            int main() {
                struct Point3D a = {1.0, 2.0, 3.0};
                struct Point3D b = {4.0, 6.0, 9.0};
                printf("Distance: %.2f\n", distance(a, b));
                return 0;
            }

             #include <stdio.h>
            #define SIZE 2
            struct Matrix {
                int mat[SIZE][SIZE];
            };
            struct Matrix multiply(struct Matrix a, struct Matrix b) {
                struct Matrix result = {0};
                for (int i = 0; i < SIZE; i++)
                    for (int j = 0; j < SIZE; j++)
                        for (int k = 0; k < SIZE; k++)
                            result.mat[i][j] += a.mat[i][k] * b.mat[k][j];
                return result;
            }
            int main() {
                struct Matrix A = {{{1, 2}, {3, 4}}};
                struct Matrix B = {{{5, 6}, {7, 8}}};
                struct Matrix C = multiply(A, B);
                for (int i = 0; i < SIZE; i++) {
                    for (int j = 0; j < SIZE; j++)
                        printf("%d ", C.mat[i][j]);
                    printf("\n");
                }
                return 0;
            }

            #include <stdio.h>
            #include <string.h>
            #define MAX_VEHICLES 100
            struct Vehicle {
                char type[20];
                char brand[20];
                float price;
            };

            int main() {
                struct Vehicle vehicles[MAX_VEHICLES];
                int n, i;
                float minPrice, maxPrice;

                printf("Enter the number of vehicles: ");
                scanf("%d", &n);
                for (i = 0; i < n; i++) {
                    printf("\nVehicle %d:\n", i + 1);
                    printf("Enter type (e.g., Car, Bike): ");
                    scanf("%s", vehicles[i].type);
                    printf("Enter brand: ");
                    scanf("%s", vehicles[i].brand);
                    printf("Enter price: ");
                    scanf("%f", &vehicles[i].price);
                }
                printf("\nEnter minimum price: ");
                scanf("%f", &minPrice);
                printf("Enter maximum price: ");
                scanf("%f", &maxPrice);
                printf("\nVehicles in the price range %.2f to %.2f:\n", minPrice, maxPrice);
                int found = 0;
                for (i = 0; i < n; i++) {
                    if (vehicles[i].price >= minPrice && vehicles[i].price <= maxPrice) {
                        printf("Type: %s, Brand: %s, Price: %.2f\n",
                            vehicles[i].type, vehicles[i].brand, vehicles[i].price);
                        found = 1;
                    }
                }

                if (!found) {
                    printf("No vehicles found in the specified price range.\n");
                }
            return 0;
            }
