#include <reg51.h>

sbit MOTOR = P1^0;

void delay(unsigned int t) {
    unsigned int i, j;
    for(i=0;i<t;i++)
        for(j=0;j<1275;j++);
}

void main() {
    while(1) {
        MOTOR = 1;   // Clockwise
        delay(1000);
    }
}

#include <reg51.h>

sfr MOTOR = 0xA0;
unsigned char code step_seq[4] = {0x09, 0x0C, 0x06, 0x03};

void delay(unsigned int t) {
    unsigned int i, j;
    for(i=0;i<t;i++)
        for(j=0;j<1275;j++);
}

void main() {
    int i;
    while(1) {
        for(i=0;i<4;i++) {
            MOTOR = step_seq[i];
            delay(500);
        }
    }
}

#include <reg51.h>

sfr ADC = 0xA0;
sbit FAN = P2^0;

void main() {
    unsigned char temp;
    while(1) {
        temp = ADC;   // Read temperature
        if(temp > 50)
            FAN = 1;  // Turn ON fan
        else
            FAN = 0;  // Turn OFF fan
    }
}
