TOPOLOGICAL SORTING

Answer: queue is never initialized with nodes of zero in-degree, so loop never starts.

def topo_dfs(graph):
    visited = set()
    stack = []
    def dfs(u):
        visited.add(u)
        for v in graph[u]:
            if v not in visited:
                dfs(v)
        stack.append(u)
    for u in graph:
        if u not in visited:
            dfs(u)
    return stack[::-1]
